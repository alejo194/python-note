 # 2
 def multi():
     L = ['1*1=1']
     while True:
        length = len(L)
        yield L
        L = ['{}*{}={}'.format(i+1,length+1,(i+1)*(length+1)) for i in range(0,length+1)]
 n = 0
 for i in multi():
     print(i)
     n += 1
     if n == 9:
         break

#### 3 裴波那契数列
```bash
def fab():
    a, b = 0, 1
    while True:
        yield b
        a, b = b, a+b
n = 0
for i in fab():
    print(i)
    n += 1
    if n == 10:
        break
```
