#### 冒泡算法
```bash
# 输入一个五位数
nums = []
while len(nums) < 5:
    num = input('input a number, please!').strip().lstrip('0')
    if not num.isdigit():
        continue
    nums.append(int(num))
length = len(nums)
count = 0
count_swap = 0
for i in range(length): #趟次
    tager = False
    count += 1
    for j in range(length - 1 - i):# 两两比较，最大的排到最后
        if nums[j] > nums[j+1]:
            nums[j], nums[j+1] = nums[j+1], nums[j]
            tager = True
            count_swap += 1
    if not tager:
        break
print("nums:{},趟次:{},一共交换多少次:{}".format(nums, count, count_swap))
```
#### 杨辉三角
```bash
def triangle():
    lis = [1]
    while True:
        length = len(lis)
        yield lis
        lis = [1] + [lis[i] + lis[i+1] for i in range(length-1)] + [1]
n = 0
for x in triangle():
    print(x)
    n += 1
    if n == 6:
        break
```

#### 字符串例题
```bash
# 输入5位数，统计每个数重复的次数
while True:
    num = input("input a number,please").strip().lstrip('+-0')
    if len(num) != 5:
        print("输入一个5位整数")
        num = input("input a number,please").strip().lstrip('+-0')
    if not num.isdigit():
        print("输入一个5位整数")
        num = input("input a number,please").strip().lstrip('+-0')
    else:
        break

l = [0]*10
for i in num: #时间复杂度O(n)
    l[int(i)]=num.count(i)
    print("字符串num:{}中数字{}重复的次数{}".format(num, i, l[(int(i))]))
    if not l[int(i)]：
       continue
```
