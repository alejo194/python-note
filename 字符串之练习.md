#### 冒泡算法
```bash
# 输入一个五位数
nums = []
while len(nums) < 5:
    num = input('input a number, please!').strip().lstrip('0')
    if not num.isdigit():
        continue
    nums.append(int(num))
length = len(nums)
count = 0
count_swap = 0
for i in range(length):
    tager = False
    count += 1
    for j in range(length - 1 - i):
        if nums[j] > nums[j+1]:
            nums[j], nums[j+1] = nums[j+1], nums[j]
            tager = True
            count_swap += 0
    if not tager:
        break
```
